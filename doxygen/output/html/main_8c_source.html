100 void SetClk24MHz() {
101  // Increase Vcore setting to level3 to support fsystem=25MHz
102  // NOTE: Change core voltage one level at a time..
103  SetVcoreUp (0x01);
104  SetVcoreUp (0x02);
105  SetVcoreUp (0x03);
106 
107  P5SEL |= BIT2+BIT3;
108  UCSCTL6 &= ~XT2OFF; // Enable XT2
109  UCSCTL6 &= ~XT2BYPASS;
110  UCSCTL3 = SELREF__XT2CLK; // FLLref = XT2
111  UCSCTL4 |= SELA_2 + SELS__DCOCLKDIV + SELM__DCOCLKDIV;
112 
113  UCSCTL0 = 0x0000; // Set lowest possible DCOx, MODx
114  // Loop until XT1,XT2 & DCO stabilizes - In this case only DCO has to stabilize
115  do
116  {
117  UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + DCOFFG); // Clear XT2,XT1,DCO fault flags
118  SFRIFG1 &= ~OFIFG; // Clear fault flags
119  } while (SFRIFG1&OFIFG); // Test oscillator fault flag
120 
121  // Disable the FLL control loop
122  __bis_SR_register(SCG0);
123 
124  // Select DCO range 24MHz operation
125  UCSCTL1 = DCORSEL_7;
126  /* Set DCO Multiplier for 24MHz
127  (N + 1) * FLLRef = Fdco
128  (5 + 1) * 4MHz = 24MHz */
129  UCSCTL2 = FLLD0 + FLLN0 + FLLN2;
130  // Enable the FLL control loop

148 void SetVcoreUp (unsigned int level)
149 {
150  // Open PMM registers for write
151  PMMCTL0_H = PMMPW_H;
152  // Set SVS/SVM high side new level
153  SVSMHCTL = SVSHE + SVSHRVL0 * level + SVMHE + SVSMHRRL0 * level;
154  // Set SVM low side to new level
155  SVSMLCTL = SVSLE + SVMLE + SVSMLRRL0 * level;
156  // Wait till SVM is settled
157  while ((PMMIFG & SVSMLDLYIFG) == 0);
158  // Clear already set flags
159  PMMIFG &= ~(SVMLVLRIFG + SVMLIFG);
160  // Set VCore to new level
161  PMMCTL0_L = PMMCOREV0 * level;
162  // Wait till new level reached
163  if ((PMMIFG & SVMLIFG))
164  while ((PMMIFG & SVMLVLRIFG) == 0);
165  // Set SVS/SVM low side to new level
166  SVSMLCTL = SVSLE + SVSLRVL0 * level + SVMLE + SVSMLRRL0 * level;
167  // Lock PMM registers for write access
168  PMMCTL0_H = 0x00;
